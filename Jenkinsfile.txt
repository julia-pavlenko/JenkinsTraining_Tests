pipeline {
   agent any
   stages { 
       
      stage('Build') {
         steps {
            git 'httpsgithub.comjulia-pavlenkoJenkinsTraining_Tests.git'
         }
      } 
      
      stage('Test') {
         steps {
            bat mvn clean test
         }
      }
      
      stage('Package') {
         steps {
            bat mvn package
         }
      }
      
      stage('Results') {
         steps {
            junit 'targetsurefire-reportsTEST-.xml'
               archiveArtifacts 'target.jar'
         }
      }
   }
}
